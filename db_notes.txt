
Table follow {
  followingUserId varchar
  followedUserId varchar
}

Table user {
  userId integer [primary key]
  email varchar
  username varchar
  password varchar
  fullName varchar
  profilePicture  varchar
  description varchar
  createdAt timestamp
  lastUpdated timestamp
}

Table post {
  postId integer [primary key]
  description text
  userId integer
  createdAt timestamp
  lastUpdated timestamp
}

Table comment {
  commentId integer [primary key]
  postId integer
  userId integer
  content text
  createdAt timestamp
  lastUpdated timestamp
}

Table reactions {
  postId integer
  userId integer
}

Table confirmInstance {
  confirmToken varchar
  username varchar
  password varchar
  email varchar
}

Table resetInstance {
  resetToken varchar
  userId integer
}

Ref: user.userId < follow.followedUserId
Ref: user.userId < follow.followingUserId
Ref: user.userId < post.userId
Ref: user.userId < comment.userId
Ref: user.userId < reactions.userId

Ref: post.postId < comment.postId
Ref: post.postId < reactions.postId

Ref: user.userId < resetInstance.userId
